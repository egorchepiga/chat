 80 /**
 81  * Нормализация текста
 82  * @param text - исходный текст
 83  * @return возвращает нормалезованный текст
 84  */
 85 function normaMessage(text) {
 86     return text
 87         .replace(/\u0020+/gim, " ")
 88         .replace(/\u0009+/gim, " ")
 89         .replace(/\u000A+/gim, '\n')
 90         .replace(/\xa0/gim, ' ')
 91         .replace(/([\u0020]*$)/gim, '')
 92         .replace(/(^[\u0020]+)/gim, '');
 93 }

 95 /**
 96  * Загрузка изображений на сервер
 97  * @param url
 98  * @param type
 99  * @return {Promise}
100  */
101 function downloadImage(url, type) {
102     return new Promise(function (resolve, reject) {
103         let date = Date.parse(new Date) + new Date().getMilliseconds(),
104             name = SHA512(url + SHA512(date + "")),
105             pathImg = `/images/${name}.${type}`,
106             options = {
107                 url: url,
108                 dest: PATH.join(__dirname, "../public/" + pathImg)
109             };
110 
111         DOWNLOAD.image(options)
112             .then(({filename, image}) => {
113                 resolve({pathImg});
114             }).catch((err) => {
115             log("image-downloader", "Ошибка при загрузке изображения", err);
116             reject(err);
117         });
118     });
119 }

121 /**
122  * проверка на ссылки картинок
123  * @param room - id комнаты
124  * @param message - текст для проверки на ссылку на картинки
125  * @param token - токен пользователя
126  * @return {Promise} впадлу уже писать. лол
127  */
128 function checkURL(room, message, token) {
129     return new Promise((resolve, reject) => {
130         let path;
131         let match = message.match(/((http(s)?)|(www\.))([^\.]+)\.([^\s]+(jpg|png))/);
132         if (!match) {
133             resolve({message: message})
134         } else {
135             message = message.replace(match[7], "");
136             downloadImage(match[0], match[7])
137                 .then(result => {
138                     path = result.pathImg;
139                     return DB.addImage(token, room, result.pathImg)
140                 }).catch(error => {
141                 log("image-downloader", "Ошибка при загрузке изображения", error);
142                 resolve({message: message});
143             }).then(result => {
144                 let ret = {
145                     path: path,
146                     id: result.id,
147                     date: result.date,
148                     message: message
149                 };
150                 resolve(ret)
151             }).catch((err) => {
152                 log("image-downloader", "Ошибка при загрузке изображения", error);
153                 resolve({message: message});
154             });
155         }
156 
157     });
158 }

160 /**
161  * Проверка входных данных (Логин, Пароль)
162  * @param data - структура входящих данных должна содержать: data.login, data.password;
163  * @return {boolean} true - если все хорошо, false - если входные данные не подходят;
164  */
165 function isReadyForLog(data) {
166     return (data.login === undefined || data.login.length < 1 || data.login.length > 20) ? false :
167         (!(data.password === undefined || data.password.length < 1 || data.password.length > 20));
168 }

172  * @param data - структура входящих данных должна содержать: data.firstName, data.lastName;
173  * @return {boolean} - true - если все хорошо, false - если входные данные не подходят;
174  */
175 function isReadyForReg(data) {
176     return (!isReadyForLog(data)) ? false :
177         (data.firstName === undefined || data.firstName.length < 1 || data.firstName.length > 20) ? false :
178             (!(data.lastName === undefined || data.lastName.length < 1 || data.lastName.length > 20));
179 }

181 /**
182  * Экранирование стрелочек, для защиты от XSS атак
183  * @param text - текст;
184  * @return {string} - текст с экранированными спецсимволами < >;
185  */
186 function accessText(text) {
187     return text
188         .replace(/[<>]/gim, function (i) {
189             return (i.charCodeAt(0) === 60) ? '<' : '>';
190         })
191 }

211 /**
212  * Добавления сообщения
213  * @param data - data.token - токен, data.idRoom - id комнаты(чата), data.message - текст сообщения
214  * @return {success: true, data: {message,from,idRoom,date,id,attachment}} attachment - вложения, в данный момент
215  * может быть image{id,from,imgPath,date}
216  */
217 function addMessage(data) {
218     return new Promise((resolve) => {
219         let message = normaMessage(data.message) || "",
220             token = data.token || "",
221             idRoom = data.idRoom || "",
222             image = {};
223         if (!message || isSpaced(message)) {
224             resolve({success: false, info: "Ошибка. Пустой текст."})
225         } else if (!idRoom) {
226             resolve({success: false, info: "Ошибка. Не выбран диалог."})
227         } else if (!token) {
228             resolve({success: false, info: "Ошибка. Токен."})
229         } else if (message.length === 0) {
230             resolve({success: false, info: "Ошибка. Пустое сообщение."})
231         } else {
232             DB.getUserInfo(token)
233                 .then(res => {
234                     const login = res.login;
235                     checkURL(idRoom, message, token)
236                         .then(result => {
237                             message = result.message;
238                             if (!isEmpty(result.path))
239                                 image = {
240                                     id: result.id,
241                                     from: login,
242                                     imgPath: result.path,
243                                     date: result.date
244                                 };
245                             return DB.addMessage(token, idRoom, accessText(message))
246                         })
247                         .then(res => {
248                             resolve({
249                                 success: true,
250                                 data: {
251                                     message: accessText(message),
252                                     from: login,
253                                     idRoom: idRoom,
254                                     date: res.date,
255                                     id: res.id,
256                                     attachment: image
257                                 },
258                                 info: "Done"
259                             });
260                         }).catch(error => {
261                         log("WARN", "Ошибка при добавлении сообщения", error);
262                         resolve({success: false, info: "Server-side error"});
263                     });
264                 })
265         }
266     })
267 }

348  */
349 function createRoom(data) {
350     return new Promise((resolve, reject) => {
351         let users;
352         if (!data.name) {
353             resolve({
354                 success: false,
355                 info: "Ошибка. Не выбрано название"
356             })
357         } else if (isEmpty(data.name)) {
358             resolve({
359                 success: false,
360                 info: "Ошибка. Не выбрано название"
361             })
362         } else if (!data.token) {
363             resolve({
364                 success: false,
365                 info: "Ошибка. Отсутствует токен"
366             })
367         } else if (isEmpty(data.token)) {
368             resolve({
369                 success: false,
370                 info: "Ошибка. Отсутствует токен"
371             })
372         } else {
373             if (typeof(data.users) === "string") {
374                 users = data.users.split(",")
375             } else {
376                 users = data.users
377             }
378             DB.createRoom(data.token, data.name, users)
379                 .then(id => {
380                     if (id !== undefined)
381                         resolve({
382                             success: true, data: {
383                                 idRoom: id,
384                                 roomName: data.name
385                             }
386                         });
387                     else {
388                         resolve({success: false, info: "Dialog is already created"});
389                     }
390                 }).catch(error => {
391                 resolve({success: false, info: "Server-side error"});
392                 log("WARN", "Ошибка при добавлении нового диалога", error);
393             });
394         }
